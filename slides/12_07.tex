\begin{frame}{Hypothesis}
\protect\hypertarget{hypothesis}{}

\((\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)\) is
a theorem of K.

\begin{itemize}
\tightlist
\item
  So we can use the base modal rules.
\end{itemize}

\end{frame}

\begin{frame}{\((\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)\)}
\protect\hypertarget{diamond-a-wedge-diamond-b-rightarrow-diamond-a-wedge-b}{}

\begin{oltableau}
[\pFmla{\False}{(\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)}{1}, just = \TAss,
]
\end{oltableau}

\bigskip

Start with it being false at 1.

\end{frame}

\begin{frame}{\((\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)\)}
\protect\hypertarget{diamond-a-wedge-diamond-b-rightarrow-diamond-a-wedge-b-1}{}

\begin{oltableau}
[\pFmla{\False}{(\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)}{1}, checked, just = \TAss,
  [\pFmla{\True}{\Diamond A \wedge \Diamond B}{1}, just = {\TRule{\False}{\rightarrow}[1]},
    [\pFmla{\False}{\Diamond (A \wedge B)}{1}, just = {\TRule{\False}{\rightarrow}[1]}
    ]
  ]
]
\end{oltableau}

\bigskip

True antecedent, false consequent.

\end{frame}

\begin{frame}{\((\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)\)}
\protect\hypertarget{diamond-a-wedge-diamond-b-rightarrow-diamond-a-wedge-b-2}{}

\begin{oltableau}
[\pFmla{\False}{(\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)}{1}, checked, just = \TAss,
  [\pFmla{\True}{\Diamond A \wedge \Diamond B}{1},  checked, just = {\TRule{\False}{\rightarrow}[1]},
    [\pFmla{\False}{\Diamond (A \wedge B)}{1},just = {\TRule{\False}{\rightarrow}[1]}
      [\pFmla{\True}{\Diamond A}{1}, just = {\TRule{\True}{\wedge}[2]}
        [\pFmla{\True}{\Diamond B}{1}, just = {\TRule{\True}{\wedge}[2]}
        ]
      ]
    ]
  ]
]
\end{oltableau}

\bigskip

Break up the and.

\end{frame}

\begin{frame}{\((\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)\)}
\protect\hypertarget{diamond-a-wedge-diamond-b-rightarrow-diamond-a-wedge-b-3}{}

\begin{oltableau}
[\pFmla{\False}{(\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)}{1}, checked, just = \TAss,
  [\pFmla{\True}{\Diamond A \wedge \Diamond B}{1}, checked,just = {\TRule{\False}{\rightarrow}[1]},
    [\pFmla{\False}{\Diamond (A \wedge B)}{1},  just = {\TRule{\False}{\rightarrow}[1]}
      [\pFmla{\True}{\Diamond A}{1}, checked, just = {\TRule{\True}{\wedge}[2]}
        [\pFmla{\True}{\Diamond B}{1}, just = {\TRule{\True}{\wedge}[2]}
          [\pFmla{\True}{A}{1.1}, just = {\TRule{\True}{\Diamond}[4]}
          ]
        ]
      ]
    ]
  ]
]
\end{oltableau}

\bigskip

Make \(\Diamond A\) true.

\end{frame}

\begin{frame}{\((\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)\)}
\protect\hypertarget{diamond-a-wedge-diamond-b-rightarrow-diamond-a-wedge-b-4}{}

\begin{oltableau}
[\pFmla{\False}{(\Diamond A \wedge \Diamond B) \rightarrow \Diamond (A \wedge B)}{1}, checked, just = \TAss,
  [\pFmla{\True}{\Diamond A \wedge \Diamond B}{1}, checked, just = {\TRule{\False}{\rightarrow}[1]},
    [\pFmla{\False}{\Diamond (A \wedge B)}{1},  just = {\TRule{\False}{\rightarrow}[1]}
      [\pFmla{\True}{\Diamond A}{1}, checked, just = {\TRule{\True}{\wedge}[2]}
        [\pFmla{\True}{\Diamond B}{1}, just = {\TRule{\True}{\wedge}[2]}
          [\pFmla{\True}{A}{1.1}, just = {\TRule{\True}{\Diamond}[4]}
            [\pFmla{\False}{A \wedge B}{1.1}, just = {\TRule{\False}{\Diamond}[3]}
            ]
          ]
        ]
      ]
    ]
  ]
]
\end{oltableau}

\bigskip

Since \(A \wedge B\) is false everywhere accessible from 1, it is false
at 1.1.

\end{frame}

\bigskip

\end{frame}

\begin{frame}{A Model}
\protect\hypertarget{a-model}{}

When the tableau doesn't close, you should be able to build a model
where the sentence is false at \(w_1\).

In fact, you should be able to read it off the tree.

\end{frame}

\begin{frame}{A Model}
\protect\hypertarget{a-model-1}{}

\begin{itemize}
\tightlist
\item
  Three worlds, \(w_1, w_{1.1}, w_{1.2}\).
\item
  The only accessibility relations are \(w_1Rw_{1.1}\) and
  \(w_1Rw_{1.2}\).
\item
  In general, the accessibility relations are the ones required by the
  restrictions (but there are none here), plus the ones required by the
  numbering, i.e., \(w_xRw_{x.y}\).
\item
  \(A\) is true at \(w_{1.1}\) and false at \(w_{1.2}\).
\item
  \(B\) is true at \(w_{1.2}\) and false at \(w_{1.1}\).
\item
  It doesn't matter what atomics are true at \(w_1\).
\item
  But both \(\Diamond A\) and \(\Diamond B\) will be true at \(w_1\),
  while \(\Diamond (A \wedge B)\) is false there.
\end{itemize}

\end{frame}
