\begin{frame}{Hypothesis}
\protect\hypertarget{hypothesis}{}

\(\Diamond A \rightarrow \Box \Diamond A\) is a theorem of S5.

\begin{itemize}
\tightlist
\item
  We are going to use simplified S5 because normal S5 is really painful
  for cases like this one.
\item
  Note that if we can prove this, we'll have shown in effect that in S5,
  all true modal claims are necessarily true. That's the distinctive
  feature of S5; modality itself is never contingent.
\end{itemize}

\end{frame}

\begin{frame}{\(\Diamond A \rightarrow \Box \Diamond A\)}
\protect\hypertarget{diamond-a-rightarrow-box-diamond-a}{}

\begin{oltableau}
[\pFmla{\False}{\Diamond A \rightarrow \Box \Diamond A}{1}, just = \TAss,
]
\end{oltableau}

\bigskip

Start with it being false at 1.

\end{frame}

\begin{frame}{\(\Diamond A \rightarrow \Box \Diamond A\)}
\protect\hypertarget{diamond-a-rightarrow-box-diamond-a-1}{}

\begin{oltableau}
[\pFmla{\False}{\Diamond A \rightarrow \Box \Diamond A}{1}, checked, just = \TAss,
  [\pFmla{\True}{\Diamond A}{1}, just = {\TRule{\False}{\rightarrow}[1]}
    [\pFmla{\False}{\Box \Diamond A}{1}, just = {\TRule{\False}{\rightarrow}[1]}
    ]
  ]
]
\end{oltableau}

\bigskip

This feels familiar by now.

\end{frame}

\begin{frame}{\(\Diamond A \rightarrow \Box \Diamond A\)}
\protect\hypertarget{diamond-a-rightarrow-box-diamond-a-2}{}

\begin{oltableau}
[\pFmla{\False}{\Diamond A \rightarrow \Box \Diamond A}{1}, checked, just = \TAss,
  [\pFmla{\True}{\Diamond A}{1}, checked, just = {\TRule{\False}{\rightarrow}[1]}
    [\pFmla{\False}{\Box \Diamond A}{1}, checked, just = {\TRule{\False}{\rightarrow}[1]}
      [\pFmla{\True}{A}{2}, just = {\TRule{\True}{\Diamond}[2]}
        [\pFmla{\False}{\Diamond A}{3}, just = {\TRule{\False}{\Box}[3]}
        ]
      ]
    ]
  ]
]
\end{oltableau}

\bigskip

\begin{itemize}
\tightlist
\item
  True \(\Diamond\) sentences and false \(\Box\) sentences have to be
  made true somewhere.
\item
  Not necessarily the same `somewhere'.
\end{itemize}

\end{frame}

\begin{frame}{\(\Diamond A \rightarrow \Box \Diamond A\)}
\protect\hypertarget{diamond-a-rightarrow-box-diamond-a-3}{}

\begin{oltableau}
[\pFmla{\False}{\Diamond A \rightarrow \Box \Diamond A}{1}, checked, just = \TAss,
  [\pFmla{\True}{\Diamond A}{1}, checked, just = {\TRule{\False}{\rightarrow}[1]}
    [\pFmla{\False}{\Box \Diamond A}{1}, checked, just = {\TRule{\False}{\rightarrow}[1]}
      [\pFmla{\True}{A}{2}, just = {\TRule{\True}{\Diamond}[2]}
        [\pFmla{\False}{\Diamond A}{3}, just = {\TRule{\False}{\Box}[3]}
          [\pFmla{\False}{A}{2}, just = {\TRule{\False}{\Diamond}[5]}]
        ]
      ]
    ]
  ]
]
\end{oltableau}

\bigskip

\begin{itemize}
\tightlist
\item
  False \(\Diamond\) sentences are false everywhere.
\item
  That is, the part inside the \(\Diamond\) is false everywhere.
\end{itemize}

\end{frame}

\begin{frame}{\(\Diamond A \rightarrow \Box \Diamond A\)}
\protect\hypertarget{diamond-a-rightarrow-box-diamond-a-4}{}

\begin{oltableau}
[\pFmla{\False}{\Diamond A \rightarrow \Box \Diamond A}{1}, checked, just = \TAss,
  [\pFmla{\True}{\Diamond A}{1}, checked, just = {\TRule{\False}{\rightarrow}[1]}
    [\pFmla{\False}{\Box \Diamond A}{1}, checked, just = {\TRule{\False}{\rightarrow}[1]}
      [\pFmla{\True}{A}{2}, just = {\TRule{\True}{\Diamond}[2]}
        [\pFmla{\False}{\Diamond A}{3}, just = {\TRule{\False}{\Box}[3]}
          [\pFmla{\False}{A}{2}, just = {\TRule{\False}{\Diamond}[5]}, close]
        ]
      ]
    ]
  ]
]
\end{oltableau}

And so it closes, so this is a theorem.

\end{frame}
